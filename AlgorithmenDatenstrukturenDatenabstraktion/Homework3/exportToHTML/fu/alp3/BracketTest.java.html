10       public static String allBrackets = "[()]"; 
11    
12       public static String getUserInput(String explanation, String inputPrefix) { 
13    
14           // we could also use System.console().readLine(), but used this to make debugging inside an IDE easier 
15    
16           //Summary:  Print explanation to the user on the console. Then print the inputPrefix and read the the 
17           //          whole line a user types and return it 
18    
19           BufferedReader reader = new BufferedReader(new InputStreamReader(System.in)); 
20           String input = ""; 
21           boolean stringParsed = false; 
22    
23           System.out.println(explanation); 
24    
25           //keep asking user for input until successful 
26           while(!stringParsed) { 
27    
28               //as with all input, we need to wrap it in a try/catch block 
29               try { 
30    
31                   System.out.print(inputPrefix); 
32                   input = reader.readLine(); 
33                   stringParsed = true; 
34               } 
35               catch(IOException e) { 
36    
37                   System.out.println("Something went wrong during input. Please try again!"); 
38               } 
39               //here we would normally close the input stream, but System.in should never be closed 
40           } 
41    
42           return input; 
43       }

45       public static boolean isAnOpeningBracket(char c) { 
46    
47           return c == '(' || c == '['; 
48       } 
49    
50       public static char getCorrespondingClosingBracket(char c) { 
51    
52           switch(c) { 
53    
54               case '[': return ']'; 
55               case '(': 
56               default : return ')'; 
57           } 
58       } 
59    
60       public static boolean isABracket(char c) { 
61    
62           //check if the given symbol is existent in the array of brackets we accept for this program 
63           return allBrackets.indexOf(c) != -1; 
64       } 
65    
66       public static boolean processChar( char symbol, Stack<Character> charStack) { 
67    
68           Character wrappedSymbol = new Character(symbol); //no longer necessary for java 5 and newer 
69    
70           //ignore all non-bracket symbols, they don't have effect on the bracket-expression 
71           if(!isABracket(symbol)) { 
72    
73               return true; 
74           } 
75    
76           //opening brackets we push to the stack 
77           if(isAnOpeningBracket(symbol)) { 
78    
79               charStack.push(wrappedSymbol); 
80           } 
81           else if(!charStack.isEmpty() && getCorrespondingClosingBracket(charStack.peek().charValue()) == symbol) { 
82    
83               //if it's a closing bracket, check if it's corresponding one is on top of the stack 
84               //if so, remove it from the stack 
85               charStack.pop(); 
86           } 
87           else { 
88    
89               //otherwise we have something similar to [), which is not a valid expression, return false 
90               //for bracket not processed successfully 
91               return false; 
92           } 
93    
94           //if we didn't explicitly return false in the else block, then everything worked out fine, 
95           //symbol processed successfully 
96           return true; 
97       } 
98    
99       public static boolean isAValidBracketExpresion(String expression) { 
100   
101          Stack<Character> charStack = new Stack<>(); 
102          char[] exprCharArray = expression.toCharArray(); 
103   
104          //go through all characters and try to process them 
105          for(char currentChar : exprCharArray) { 
106   
107              //if symbol was not processed correctly, break the loop and return false immediately, 
108              //no need to continue checking 
109              if(!processChar(currentChar, charStack)) return false; 
110          } 
111   
112          //if we successfully processed each character, check if the bracket-stack is empty, if so, 
113          //every bracket has its corresponding closing bracket in the correct place, expression is valid 
114          return charStack.isEmpty(); 
115      }

117      public static void printResult(boolean isValid) { 
118   
119          System.out.println("This is " + 
120                              (isValid ? "" : "NOT") + 
121                              " a valid bracket expression!"); 
122      }

124      public static void main(String[] args) { 
125   
126          String bracketExpression = getUserInput("Enter a bracket expression to see if it's valid.", 
127                  "Bracket expression: "); 
128          boolean isValid = isAValidBracketExpresion(bracketExpression); 
129   
130          printResult(isValid);
