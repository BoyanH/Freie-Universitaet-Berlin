7    public class BinTree<E> {

9         * reference to the root:

11       private BTNode<E> root; 
12       /** 
13        * number of nodes: 
14        */ 
15       private int size; 
16    
17       /** 
18        * default constructor: 
19        */ 
20       public BinTree() {

22           size = 1;

30           return size; 
31       }

34           return (size == 1); 
35       }

38        * a trivial Tree consists of the root node only 
39        * empty trees are excluded 
40        */ 
41    
42       public boolean isInternal(BTNode<E> v) { 
43           return v.isInternal(); 
44       }

47           return v.isLeaf(); 
48       }

63           return v.rightChild(); 
64       } 
65    
66       public BTNode<E> sibling(BTNode<E> v) { 
67           BTNode<E> p = parent(v); 
68           BTNode<E> lc = leftChild(p); 
69           if (v == lc) 
70               return rightChild(p); 
71           else 
72               return lc; 
73       } 
74    
75       public BTNode<E> parent(BTNode<E> v) { 
76           return v.parent(); 
77       }

80    
81       /** 
82        * turn a leaf v to an inner node by appending two new leaves 
83        * if v is not a leaf, the method won't do anything 
84        * alternatively one could throw an exception 
85        */ 
86       public void expandExternalNode(BTNode<E> v) { 
87           if (isLeaf(v)) { 
88               v.setLeft(new BTNode<E>(null, v, null, null)); 
89               v.setRight(new BTNode<E>(null, v, null, null)); 
90               size += 2; 
91           } 
92       } 
93    
94       /** 
95        * delete a leaf v and replace v's parent node by v's sibling 
96        * if v is the root or if v is not a leaf, the method won't do anything 
97        * alternatively one could throw an exception 
98        */ 
99       public void removeAboveExternalNode(BTNode<E> v) { 
100          if (isLeaf(v) && !isRoot(v)) { 
101              BTNode<E> p = parent(v); 
102              BTNode<E> s = sibling(v); 
103              if (isRoot(p)) { 
104                  s.setParent(null); 
105                  root = s; 
106              } else { 
107                  BTNode<E> g = parent(p); 
108                  if (p == leftChild(g)) 
109                      g.setLeft(s); 
110                  else 
111                      g.setRight(s); 
112                  s.setParent(g); 
113              } 
114              size -= 2; 
115          } 
116      }
