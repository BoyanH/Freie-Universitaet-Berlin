10       private static int getNumberOfEdgesInG(Integer[][] adj) { 
11    
12           /* 
13               Summary: count only the edges (u,v) where u > v, as in a non-directional graph 
14               the same edges are found twice in the adjacency list 
15            */ 
16    
17           int count = 0; 
18    
19           for(int i = 0; i < adj.length; i++) { 
20    
21    
22               for(int j = 0; j < adj[i].length; j++) { 
23    
24                   //Count only edges {(u, v) | u <= v} as in an unidirectional graph, same edges are found twice in the 
25                   //adjacency list 
26                   if(adj[i][j] > i) { 
27    
28                       count++; 
29                   } 
30               } 
31           } 
32    
33           return count; 
34       }

36       public static LinkedList<LinkedList<Integer>> getLinearGraphEdgeNumbering(Integer[][] adj) { 
37    
38           /* 
39               Summary: returns the adjacency list of the linear graph from the given one, 
40                       where all edges are already replaced with their numbers 
41            */ 
42    
43           //P.S.: Please don't say again that the comments are copied, I am adding them in the last minute ^.^ 
44    
45           //we could always add nested LinkedLists to linearEdgeNumbering when the index is not available, 
46           //this just speeds things up and better shows how everything works 
47           int numberOfEdges = LinearGraphEdgeNumbering.getNumberOfEdgesInG(adj); 
48           LinkedList<LinkedList<Integer>> linearEdgeNumbering = new LinkedList<>(); 
49    
50           for(int i = 0; i < numberOfEdges; i++) { 
51    
52               linearEdgeNumbering.add(new LinkedList<>()); 
53           } 
54    
55           for(int i = 0; i < adj.length; i++) { 
56    
57    
58               for(int j = 0; j < adj[i].length; j++) { 
59    
60                   //find every adjacency pair (vertex -> adjacency list -> nth adjacent vertex) 
61    
62                   //get it's numbering for later on 
63                   Integer currentLinearGraphVertexNumber = getEdgeNumber(adj, new Integer[]{i, adj[i][j]}); 
64    
65                   int u = adj[i][j]; 
66    
67                   //get all adjacent edges, or said otherwise - all edges that are between the current adjacent 
68                   //vertex and it's adjacent vertexes 
69                   for(int l = 0; l < adj[u].length; l++) { 
70    
71                       int v = adj[u][l]; 
72    
73                       //get the numbering of the current "adjacent edge" (explained on line 67, 68) 
74                       Integer currentLinearGraphEdgeNumber = getEdgeNumber(adj, new Integer[]{u, v}); 
75                       if(linearEdgeNumbering.get(currentLinearGraphVertexNumber) == null) { 
76    
77                           //if this is the first adjacent edge we find, initialize the linked list 
78                           linearEdgeNumbering.set(currentLinearGraphVertexNumber, new LinkedList<>()); 
79                       } 
80    
81                       //if we didn't already find and add this edge to the adjacency list (happens as graph is unidirectional) 
82                       if(!linearEdgeNumbering.get(currentLinearGraphVertexNumber).contains(currentLinearGraphEdgeNumber)) { 
83    
84                           //add the new edge's numbering and sort the list again to keep things clear 
85                           linearEdgeNumbering.get(currentLinearGraphVertexNumber).add(currentLinearGraphEdgeNumber); 
86                           linearEdgeNumbering.get(currentLinearGraphVertexNumber).sort(new Comparator<Integer>() { 
87    
88                               @Override 
89                               public int compare(Integer a, Integer b) { 
90    
91                                   return a - b; 
92                               } 
93                           }); 
94                       } 
95                   } 
96               } 
97           } 
98    
99           return linearEdgeNumbering; 
100      }

103   
104          /* 
105   
106             We developed an algorithm to get the whole linear graph of the given one as adjacency list 
107             and replace each edge with its numbering

118      public static int getEdgeNumber(Integer[][] adj, Integer[] e) { 
119   
120          int counter = -1; //to start counting from 0 not 1 
121   
122          for(int i = 0; i < adj.length; i++) { 
123   
124              for(int j = 0; j < adj[i].length; j++) { 
125   
126                  if(adj[i][j] > i) { 
127   
128                      counter++; 
129                  } 
130                  if(e[0] == i && e[1] == adj[i][j] || 
131                          e[0] == adj[i][j] && e[1] == i) 
132                      return counter; 
133              } 
134          } 
135   
136          return counter; 
137      }
